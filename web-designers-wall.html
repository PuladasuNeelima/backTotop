Java Script

<script>
  (function(){
    $("body").backTotop({
      isWindow: true,
      text: "BACK TO TOP <i class='icon-circle-arrow-up icon-2x'></i>",
      cssClass:"web-designer-wall",
      buttonPos:{
        left: 20,
        bottom: 20
      }
    });
  })();
</script>
