Java Script

<script>
  (function(){
     $("body").backTotop({
          isWindow: true,
          text: "<i class='icon-circle-arrow-up icon-3x'></i>",
          cssClass:"flipkart",
          buttonPos:{
            bottom: 50,
            right: 0
          }
      });
  })();
</script>
